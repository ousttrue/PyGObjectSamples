$ gst-launch-1.0 videotestsrc ! xvimagesink

上記コマンドラインをpythonにする。

#!/usr/bin/python3
# coding: utf-8

import gi
gi.require_version('Gst', '1.0')
from gi.repository import GObject, Gst


def create_pipeline():
    pipeline = Gst.Pipeline()

    src = Gst.ElementFactory.make('videotestsrc', 'source')
    pipeline.add(src)
    assert src 

    sink = Gst.ElementFactory.make('xvimagesink', 'sink')
    pipeline.add(sink)
    assert sink

    src.link(sink)

    return pipeline

if __name__=="__main__":
    Gst.init(None)
    GObject.threads_init()

    pipeline=create_pipeline()
    pipeline.set_state(Gst.State.PLAYING)
    GObject.MainLoop().run()

#!/usr/bin/python3
# coding: utf-8

import gi
gi.require_version('Gtk', '3.0')
gi.require_version('GdkX11', '3.0') # required for get_xid
gi.require_version('Gst', '1.0')
gi.require_version('GstVideo', '1.0') # required for set_window_handle
from gi.repository import Gtk, GdkX11, Gst, GstVideo
import os, sys


class GtkOverlayVideo:
    def __init__(self, path, get_xid):
        self.get_xid=get_xid

        # Setup GStreamer 
        self.player = Gst.ElementFactory.make("playbin", None)
        self.player.set_property("uri", "file://"+path)

        bus = self.player.get_bus() 
        bus.add_signal_watch() 
        bus.enable_sync_message_emission() 
        #used to get messages that GStreamer emits 
        bus.connect("message", self.on_message) 
        #used for connecting video to your application 
        bus.connect("sync-message::element", self.on_sync_message)
        self.player.set_state(Gst.State.PLAYING)

    def on_message(self, bus, message): 
        if message.type == Gst.MessageType.EOS: 
            # End of Stream 
            self.player.seek(1.0, Gst.Format.TIME, 
                    Gst.SeekFlags.FLUSH, 
                    Gst.SeekType.SET, 5000000000, 
                    Gst.SeekType.NONE, 6000000000)
        elif message.type == Gst.MessageType.ERROR: 
            self.player.set_state(Gst.State.NULL) 
            (err, debug) = message.parse_error() 
            print("Error: %s %s)" % (err, debug))

    def on_sync_message(self, bus, message):
        s=message.get_structure()
        if not s:
            return
        name=s.get_name()
        if name != "prepare-window-handle":
            return
        xid=self.get_xid()
        if not xid:
            return
        imagesink = message.src
        imagesink.set_property("force-aspect-ratio", True)
        imagesink.set_window_handle(xid)



class VideoWindow(Gtk.Window):
    def __init__(self):
        super().__init__()
        self.connect("delete-event", Gtk.main_quit)
        self.connect("realize", self.on_realize)

    def on_realize(self, sender):
        self.xid = sender.get_property('window').get_xid()

    def get_xid(self):
        return self.xid


if __name__ == "__main__":
    Gst.init(None)

    window=VideoWindow()
    video=GtkOverlayVideo(sys.argv[1], window.get_xid)

    window.show_all()
    Gtk.main()
