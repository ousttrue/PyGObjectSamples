HorizontalBox¶

#!/usr/bin/python3
# coding: utf-8

import gi
gi.require_version('Gtk', '3.0')
from gi.repository import Gtk


class HorizBox(Gtk.Window):
    def __init__(self, title="HorizBox"):
        super().__init__(title=title)
        self.connect("delete-event", Gtk.main_quit)

        box=Gtk.Box()
        self.add(box)

        button1=Gtk.Button(label="Button1")
        box.pack_start(button1, expand=True, fill=True, padding=0)

        button2=Gtk.Button(label="Button2")
        box.pack_start(button2, expand=True, fill=True, padding=0)


if __name__=="__main__":
    win=HorizBox()
    win.show_all()
    Gtk.main()
